Add in the end of plugin.xml of the OntoUML.diagram to enable new buttons in mouse click:

      <extension point="org.eclipse.ui.bindings">
      <key commandId="OntoUML.createKind" sequence="M1+I" schemeId="org.eclipse.ui.defaultAcceleratorConfiguration"/>
   </extension>
   
   <extension point="org.eclipse.ui.commands">
      <category name="OntoUML" description="Commands related to OntoUML diagrams." id="category.OntoUML"/>
      <command categoryId="category.OntoUML" description="Creates a Kind" id="OntoUML.createKind" name="Create Kind">
      </command>
   </extension>
   
   <extension point="org.eclipse.ui.popupMenus">
      <objectContribution
         adaptable="false"
         id="OntoUML.diagram.ui.objectContribution.KindEditPart1"
         objectClass="OntoUML.diagram.edit.parts.KindEditPart">
         <menu 
            id="OntoUMLCreate" 
            label="&amp;Create" 
            path="additions"> 
            <separator name="group1"/>
         </menu>
         <action
            class="OntoUML.diagram.part.OntoUMLCreateKindAction"
            definitionId="OntoUML.createKind"
            enablesFor="1"
            id="OntoUML.popup.OntoUMLCreateKindActionID"
            label="&amp;Kind"
            menubarPath="OntoUMLCreate/group1">
         </action>
      </objectContribution>                    
   </extension>
